1. Вивести значення наступних колонок: номер, код, новинка, назва, ціна, сторінки

SELECT num AS 'Номер',code AS 'Код книги',is_new AS 'Новинка', name AS 'Назва книги',price AS 'Ціна', publisher 'Видавництво'
FROM book;

2. Вивести значення всіх колонок

SELECT * FROM book;

3. Вивести значення колонок в наступному порядку: код, назва, новинка, сторінки, ціна, номер

SELECT 	code AS 'Код', name AS 'Назва', is_new AS 'Новинка', pages AS 'Сторінки', price AS 'Ціна',num AS 'Номер' FROM book;

4. Вивести значення всіх колонок 10 перших рядків

SELECT * FROM book LIMIT 10;

5. Вивести значення всіх колонок 10% перших рядків

SELECT TOP 10 PERCENT * FROM book;

6. Вивести значення колонки код без повторення однакових кодів

SELECT DISTINCT code FROM book;

7. Вивести всі книги новинки

SELECT code AS 'Код', name AS 'Назва',publisher AS 'Видавництво',price AS 'Ціна' FROM book WHERE is_new='Yes'; 

8. Вивести всі книги новинки з ціною від 20 до 30

SELECT code AS 'Код', name AS 'Назва',publisher AS 'Видавництво',price AS 'Ціна' FROM book WHERE is_new='Yes' AND price BETWEEN 20 AND 30; 

9. Вивести всі книги новинки з ціною менше 20 і більше 30

SELECT code AS 'Код', name AS 'Назва',publisher AS 'Видавництво',price AS 'Ціна' FROM book WHERE is_new='Yes' AND price NOT BETWEEN 20 AND 30; 

10. Вивести всі книги з кількістю сторінок від 300 до 400 і з ціною більше 20 і менше 30

SELECT code AS 'Код', name AS 'Назва',publisher AS 'Видавництво',price AS 'Ціна' FROM book WHERE pages BETWEEN 300 AND 400 AND price BETWEEN 20 AND 30; 

11. Вивести всі книги видані взимку 2000 року

SELECT code AS 'Код', name AS 'Назва',publisher AS 'Видавництво',price AS 'Ціна',date AS 'Дата' FROM book WHERE date BETWEEN CAST('01/01/2000' AS DATE) AND CAST('29/02/2000' AS DATE) OR date BETWEEN CAST('01/12/2000' AS DATE) AND CAST('31/12/2000' AS DATE)

12. Вивести книги зі значеннями кодів 5110, 5141, 4985, 4241

SELECT * FROM book WHERE code IN(5110, 5141, 4985, 4241);

13. Вивести книги видані в 1999, 2001, 2003, 2005 р

SELECT * FROM book WHERE YEAR(date) IN(1999,2001,2003,2005);

14. Вивести книги назви яких починаються на літери А-К

SELECT * FROM book WHERE name >='А' AND name <='К';

15. Вивести книги назви яких починаються на літери "АПП", видані в 2000 році з ціною до 20

SELECT * FROM book WHERE name LIKE 'АПП%' AND date LIKE '2000%' AND price <20;

16. Вивести книги назви яких починаються на літери "АПП", закінчуються на "е", видані в першій половині 2000 року

SELECT * FROM book WHERE name LIKE 'АПП%Е' AND date BETWEEN CAST('2000-01-01' AS DATE) AND CAST('2000-06-30' AS DATE);

17. Вивести книги, в назвах яких є слово Microsoft, але немає слова Windows

SELECT * FROM book WHERE name LIKE '%Microsoft%' AND name NOT LIKE '%Windows%';

18. Вивести книги, в назвах яких присутня як мінімум одна цифра.

SELECT * FROM book WHERE name RLIKE '[0-9]';

19. Вивести книги, в назвах яких присутні не менше трьох цифр.

SELECT * FROM book WHERE name RLIKE '.*[0-9].*[0-9].*[0-9].*';
