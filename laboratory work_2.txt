1. Вивести значення наступних колонок: номер, код, новинка, назва, ціна, сторінки

SELECT Num AS 'Номер', Code AS 'Код', New AS 'Новинка', Name AS 'Назва', Price AS 'Ціна', Pages AS 'Сторінки'
FROM books1;

2. Вивести значення всіх колонок

SELECT * FROM books1;

3. Вивести значення колонок в наступному порядку: код, назва, новинка, сторінки, ціна, номер

SELECT 	Code AS 'Код', Name AS 'Назва', New AS 'Новинка', Pages AS 'Сторінки', Price AS 'Ціна', Num AS 'Номер' 
FROM books1;

4. Вивести значення всіх колонок 10 перших рядків

SELECT * FROM books1 LIMIT 10;

5. Вивести значення всіх колонок 10% перших рядків

SELECT TOP 10 PERCENT * FROM books1;

6. Вивести значення колонки код без повторення однакових кодів

SELECT DISTINCT Code FROM books1;

7. Вивести всі книги новинки

SELECT Num AS 'Номер', Code AS 'Код', New AS 'Новинка', Name AS 'Назва', Price AS 'Ціна', Publisher AS 'Видавництво', Pages AS 'Сторінки', Format AS 'Формат',
Date AS 'Дата', Circulation AS 'Тираж', Category AS 'Категорія'
FROM books1 WHERE New = 'Yes';

8. Вивести всі книги новинки з ціною від 20 до 30

SELECT Num AS 'Номер', Code AS 'Код', New AS 'Новинка', Name AS 'Назва', Price AS 'Ціна', Publisher AS 'Видавництво', Pages AS 'Сторінки', Format AS 'Формат',
Date AS 'Дата', Circulation AS 'Тираж', Category AS 'Категорія'
FROM books1 WHERE New = 'Yes' AND Price BETWEEN 20 AND 30;

9. Вивести всі книги новинки з ціною менше 20 і більше 30

SELECT Num AS 'Номер', Code AS 'Код', New AS 'Новинка', Name AS 'Назва', Price AS 'Ціна', Publisher AS 'Видавництво', Pages AS 'Сторінки', Format AS 'Формат',
Date AS 'Дата', Circulation AS 'Тираж', Category AS 'Категорія'
FROM books1 WHERE New = 'Yes' AND Price NOT BETWEEN 20 AND 30;

10. Вивести всі книги з кількістю сторінок від 300 до 400 і з ціною більше 20 і менше 30

SELECT Num AS 'Номер', Code AS 'Код', New AS 'Новинка', Name AS 'Назва', Price AS 'Ціна', Publisher AS 'Видавництво', Pages AS 'Сторінки', Format AS 'Формат',
Date AS 'Дата', Circulation AS 'Тираж', Category AS 'Категорія'
FROM books1 WHERE Pages BETWEEN 300 AND 400 AND Price BETWEEN 20 AND 30;

11. Вивести всі книги видані взимку 2000 року

SELECT Num AS 'Номер', Code AS 'Код', New AS 'Новинка', Name AS 'Назва', Price AS 'Ціна', Publisher AS 'Видавництво', Pages AS 'Сторінки', Format AS 'Формат',
Date AS 'Дата', Circulation AS 'Тираж', Category AS 'Категорія'
FROM books1 WHERE Date BETWEEN CAST('2000/01/01' AS DATE) AND CAST('2000/02/29' AS DATE) OR Date BETWEEN CAST('2000/12/01' AS DATE) AND CAST('2000/12/31' AS DATE);

12. Вивести книги зі значеннями кодів 5110, 5141, 4985, 4241

SELECT * FROM books1 WHERE Code IN(5110, 5141, 4985, 4241);

13. Вивести книги видані в 1999, 2001, 2003, 2005 р

SELECT * FROM books1 WHERE YEAR(Date) IN(1999,2001,2003,2005);

14. Вивести книги назви яких починаються на літери А-К

SELECT * FROM books1 WHERE Name>='А' AND Name<='К';

15. Вивести книги назви яких починаються на літери "АПП", видані в 2000 році з ціною до 20

SELECT * FROM books1 WHERE Name LIKE 'АПП%' AND Date LIKE '2000%' AND Price<20;

16. Вивести книги назви яких починаються на літери "АПП", закінчуються на "е", видані в першій половині 2000 року

SELECT * FROM books1 WHERE Name LIKE 'АПП%Е' AND Date BETWEEN CAST('2000-01-01' AS DATE) AND CAST('2000-06-30' AS DATE);

17. Вивести книги, в назвах яких є слово Microsoft, але немає слова Windows

SELECT * FROM books1 WHERE Name LIKE '%Microsoft%' AND Name NOT LIKE '%Windows%';

18. Вивести книги, в назвах яких присутня як мінімум одна цифра.

SELECT * FROM books1 WHERE Name RLIKE '[0-9]';

19. Вивести книги, в назвах яких присутні не менше трьох цифр.

SELECT * FROM books1 WHERE Name RLIKE '.*[0-9].*[0-9].*[0-9].*';
